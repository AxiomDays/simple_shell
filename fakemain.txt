int main (void)
{
	int i = 0;
	char **contain;

	contain = tok_array(array);

		i++;
	}

char** tok_array(char* input)
{
        int i = 0;
        char *tok;
        char **arr = NULL;
        char *buf = NULL;

        buf = strdup(input);
        tok = strtok(buf, " \n");
        while (tok)
        {
                i++;
                tok = strtok(NULL, " \n");
        }
        arr = (char **)malloc(sizeof(char *) * (i + 1));
        if (!arr)
        {
                print_err_mes();
                free_2d(arr);
                exit(errno);
        }
        strcpy(buf, input);
        i = 0;
        tok = strtok(buf, " \n");
        while (tok)
        {
                arr[i] = strdup(tok);
                if (!arr[i])
                {
                        print_err_mes();
                        free_2d(arr);
                        exit(errno);
                }
                tok = strtok(NULL, " \n");
                i++;
        }
        free(buf);
        return (arr);
}
